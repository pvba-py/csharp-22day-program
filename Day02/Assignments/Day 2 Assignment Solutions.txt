SELECT 
p.FullName as ProviderName, d.Name as DepartmentName,
COUNT(*) as TotalEncounters, DENSE_RANK() OVER (
        ORDER BY COUNT(*) DESC
    ) AS ProviderRank
From Provider p
LEFT JOIN Department d ON p.DepartmentId = d.DepartmentId
JOIN Encounter e ON p.ProviderId = e.ProviderId
GROUP BY p.ProviderId,p.FullName,d.Name;

3. Assignment Question 3

Ans:

Billing View:

CREATE OR ALTER VIEW view_BillingClaims
AS
SELECT
    c.ClaimID,
    i.PatientId,
    c.Status,
    c.BilledAmount,
    c.ReimbursedAmt,
    i.EffectiveDate
FROM Insurance i
JOIN Claim c ON i.InsuranceId = c.InsuranceId;
GO

Stored Procedure to identify where revenue recovery efforts should focus:

CREATE PROCEDURE storedProcedure_RevenueLeakageReport
AS
BEGIN
	SET NOCOUNT ON;
	SELECT
        Status,
        COUNT(*) AS TotalClaims,
        SUM(BilledAmount) AS TotalBilledAmount,
        SUM(ReimbursedAmt) AS TotalReimbursedAmount,
        SUM(BilledAmount - ReimbursedAmt) AS OutstandingAmount,

        RANK() OVER (
            ORDER BY SUM(BilledAmount - ReimbursedAmt) DESC
        ) AS LossRank

    FROM view_BillingClaims
    GROUP BY Status
    ORDER BY OutstandingAmount DESC;
END;
GO

2. Provider Workload

CREATE PROCEDURE sp_ProviderWorkload
AS
BEGIN
    SELECT
        ProviderID,
        COUNT(*) AS TotalVisits
    FROM Encounter
    GROUP BY ProviderID;
END
GO

3. High Risk Patients

CREATE PROCEDURE sp_HighRiskPatients
AS
BEGIN
    SELECT
        PatientId,
        COUNT(*) AS EmergencyVisits
    FROM Encounter
    WHERE EncounterType = 'ED'
    GROUP BY PatientId
    ORDER BY EmergencyVisits DESC;
END;
GO

4. 30 Day Readmissions

CREATE PROCEDURE sp_30DayReadmissions
AS
BEGIN
    WITH PatientEncounters AS
    (
        SELECT
            EncounterId,
            PatientId,
            AdmitDate,
            DischargeDate,
            LAG(DischargeDate) OVER
            (
                PARTITION BY PatientId
                ORDER BY AdmitDate
            ) AS PreviousDischargeDate
        FROM Encounter
    )

    SELECT
        EncounterId,
        PatientId,
        PreviousDischargeDate,
        AdmitDate,
        DATEDIFF(DAY, PreviousDischargeDate, AdmitDate) AS DaysBetweenVisits
    FROM PatientEncounters
    WHERE PreviousDischargeDate IS NOT NULL
      AND DATEDIFF(DAY, PreviousDischargeDate, AdmitDate) <= 30;
END;
GO

CREATE VIEW view_Clinical
AS
SELECT
    p.PatientId,
    p.FullName,
    e.EncounterId,
    e.EncounterType,
    e.AdmitDate,
    e.DischargeDate,
	d.Description,
    d.DiagnosedOn
FROM Patient p
JOIN Encounter e
    ON p.PatientId = e.PatientId
JOIN Diagnosis d
    ON e.EncounterId = d.EncounterId;
GO

3. Analytical View

CREATE VIEW view_Analytics
AS
SELECT
	CASE
		WHEN DATEDIFF(YEAR,p.DateOfBirth,GETDATE()) < 18 THEN '0-17'
		WHEN DATEDIFF(YEAR,p.DateOfBirth,GETDATE()) BETWEEN 18 AND 35 THEN '18-35'
		WHEN DATEDIFF(YEAR,p.DateOfBirth,GETDATE()) BETWEEN 36 AND 55 THEN '36-55'
		ELSE '56+'
	END AS AgeBand,
	e.EncounterType,
	e.DepartmentId
FROM Encounter e
JOIN Patient p
	ON e.PatientId = p.PatientId;
GO
